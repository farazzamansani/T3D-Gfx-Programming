63  Quaternion(float theta_x, float theta_y, float theta_z)
64  {
65  float cos_z_2 = cosf(0.5f*theta_z);
66  float cos_y_2 = cosf(0.5f*theta_y);
67  float cos_x_2 = cosf(0.5f*theta_x);
68 
69  float sin_z_2 = sinf(0.5f*theta_z);
70  float sin_y_2 = sinf(0.5f*theta_y);
71  float sin_x_2 = sinf(0.5f*theta_x);
72 
73  // and now compute Quaternion
74  s = cos_z_2*cos_y_2*cos_x_2 + sin_z_2*sin_y_2*sin_x_2;
75  v.x = cos_z_2*cos_y_2*sin_x_2 - sin_z_2*sin_y_2*cos_x_2;
76  v.y = cos_z_2*sin_y_2*cos_x_2 + sin_z_2*cos_y_2*sin_x_2;
77  v.z = sin_z_2*cos_y_2*cos_x_2 - cos_z_2*sin_y_2*sin_x_2;
78  }

103  Quaternion(const Matrix3x3 &m)
104  {
105  float t = m[0][0]+m[1][1]+m[2][2];
106  float r = sqrt(1+t);
107  s = 0.5f*r;
108  v.x = 0.5f*sqrt(1+m[0][0]-m[1][1]-m[2][2]);
109  if ((v.x>0 && m[2][1]-m[1][2]<0) || (v.x<0 && m[2][1]-m[1][2]>0)) v.x = -v.x;
110  v.y = 0.5f*sqrt(1-m[0][0]+m[1][1]-m[2][2]);

293  static Quaternion slerp(const Quaternion &q1, const Quaternion &q2, float t)
294  {
295  Quaternion q3;
296  float dot = Quaternion::dot(q1, q2);
297 
298  /* dot = cos(theta)
299  if (dot < 0), q1 and q2 are more than 90 degrees apart,
300  so we can invert one to reduce spinning */
301  if (dot < 0)
302  {
303  dot = -dot;
304  q3 = -q2;
305  } else q3 = q2;
306 
307  if (dot < 0.95f)
308  {
309  float angle = acosf(dot);
310  return (q1*sinf(angle*(1-t)) + q3*sinf(angle*t))/sinf(angle);
311  } else // if the angle is small, use linear interpolation
312  return lerp(q1,q3,t);
313  }

355  Vector3 eulerAngles() const
356  {
357  float sqw = s*s;
358  float sqx = v.x*v.x;
359  float sqy = v.y*v.y;
360  float sqz = v.z*v.z;
361 
362  float unit = sqx + sqy + sqz + sqw;
363  float test = v.x * v.y + v.z * s;
364 
365  Vector3 euler;
366 
367  if (test > 0.4999f * unit) // Singularity at north pole
368  {
369  euler.y = 2.f * (float)atan2f(v.x, s); // Yaw
370  euler.z = Math::PI * 0.5f; // Roll
371  euler.x = 0.f; // Pitch
372  return euler;
373  }
374  else if (test < -0.4999f * unit) // Singularity at south pole
375  {
376  euler.y = -2.f * (float)atan2f(v.x, s); // Yaw
377  euler.z = -Math::PI * 0.5f; // Roll
378  euler.x = 0.f; // Pitch
379  return euler;
380  }
381  else
382  {
383  euler.y = (float)atan2f(2.f * v.y * s - 2.f * v.x * v.z, sqx - sqy - sqz + sqw); // Yaw
384  euler.z = (float)asinf(2.f * test / unit); // Roll
385  euler.x = (float)atan2f(2.f * v.x * s - 2.f * v.y * v.z, -sqx + sqy - sqz + sqw); // Pitch
386  }
387  return euler;
388  }
