104  inline Matrix4x4 concatenate(const Matrix4x4 &m2) const
105  {
106  Matrix4x4 r;

108  r.m[0][1] = m[0][0] * m2.m[0][1] + m[0][1] * m2.m[1][1] + m[0][2] * m2.m[2][1] + m[0][3] * m2.m[3][1];

110  r.m[0][3] = m[0][0] * m2.m[0][3] + m[0][1] * m2.m[1][3] + m[0][2] * m2.m[2][3] + m[0][3] * m2.m[3][3];
111 
112  r.m[1][0] = m[1][0] * m2.m[0][0] + m[1][1] * m2.m[1][0] + m[1][2] * m2.m[2][0] + m[1][3] * m2.m[3][0];
113  r.m[1][1] = m[1][0] * m2.m[0][1] + m[1][1] * m2.m[1][1] + m[1][2] * m2.m[2][1] + m[1][3] * m2.m[3][1];
114  r.m[1][2] = m[1][0] * m2.m[0][2] + m[1][1] * m2.m[1][2] + m[1][2] * m2.m[2][2] + m[1][3] * m2.m[3][2];
115  r.m[1][3] = m[1][0] * m2.m[0][3] + m[1][1] * m2.m[1][3] + m[1][2] * m2.m[2][3] + m[1][3] * m2.m[3][3];

146  inline Vector3 operator * ( const Vector3 &v ) const
147  {
148  Vector3 r;
149 
150  float fInvW = 1.0f / ( m[3][0] * v.x + m[3][1] * v.y + m[3][2] * v.z + m[3][3] );
151 
152  r.x = ( m[0][0] * v.x + m[0][1] * v.y + m[0][2] * v.z + m[0][3] ) * fInvW;
153  r.y = ( m[1][0] * v.x + m[1][1] * v.y + m[1][2] * v.z + m[1][3] ) * fInvW;
154  r.z = ( m[2][0] * v.x + m[2][1] * v.y + m[2][2] * v.z + m[2][3] ) * fInvW;
155 
156  return r;
157  }

169  inline Plane operator * (const Plane& p) const
170  {
171  Plane ret;
172  Matrix4x4 invTrans = inverse().transpose();
173  Vector4 v4( p.normal.x, p.normal.y, p.normal.z, p.d );
174  v4 = invTrans * v4;
175  ret.normal.x = v4.x;
176  ret.normal.y = v4.y;
177  ret.normal.z = v4.z;
178  ret.d = v4.w / ret.normal.length();
179  ret.normal.normalise();
180 
181  return ret;
